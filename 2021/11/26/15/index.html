return courseName;
}

return scores;

public void sort(){
for(int i = 0;i<scores.length-1;i++){
for(int j = 0;j<scores.length-1;j++){
if(scores[j]>scores[j+1]){
float temp = scores[j];
scores[j] =scores[j+1];
scores[j+1]=temp;

}
}
